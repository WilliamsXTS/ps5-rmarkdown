LE_data <- gapminder %>% 
  filter(time %in% c(1960, 2019)) %>% 
  select(name, time, region, lifeExpectancy)

LE_growth <- LE_data %>% 
  group_by(name) %>% 
  mutate(growth = lifeExpectancy - lag(lifeExpectancy)) %>% 
  filter(!is.na(growth)) %>%
